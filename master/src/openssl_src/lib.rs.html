63    fn cmd_make(&self) -> Result<Command, &'static str> {
64        let host = &self.host.as_ref().ok_or("HOST dir not set")?[..];
65        Ok(
66            if host.contains("dragonfly")
67                || host.contains("freebsd")
68                || host.contains("openbsd")
69                || host.contains("solaris")

79    #[cfg(windows)]
80    fn check_env_var(&self, var_name: &str) -> Option<bool> {
81        env::var_os(var_name).and_then(|s| {
82            if s == "1" {
83                // a message to stdout, let user know asm is force enabled
84                println!(
85                    "cargo:warning={}: nasm.exe is force enabled by the \
86                    'OPENSSL_RUST_USE_NASM' env var.",
87                    env!("CARGO_PKG_NAME")
88                );
89                Some(true)
90            } else if s == "0" {
91                // a message to stdout, let user know asm is force disabled
92                println!(
93                    "cargo:warning={}: nasm.exe is force disabled by the \
94                    'OPENSSL_RUST_USE_NASM' env var.",
95                    env!("CARGO_PKG_NAME")
96                );
97                Some(false)
98            } else {
99                println!(
100                    "cargo:warning=The environment variable {} is set to an unacceptable value: {:?}",
101                    var_name, s
102                );
103                None
104            }
105        })
106    }

108    #[cfg(windows)]
109    fn is_nasm_ready(&self) -> bool {
110        self.check_env_var("OPENSSL_RUST_USE_NASM")
111            .unwrap_or_else(|| {
112                // On Windows, use cmd `where` command to check if nasm is installed
113                Command::new("cmd")
114                    .args(&["/C", "where nasm"])
115                    .output()
116                    .map(|w| w.status.success())
117                    .unwrap_or(false)
118            })

127    /// Exits the process on failure. Use `try_build` to handle the error.
128    pub fn build(&mut self) -> Artifacts {
129        match self.try_build() {
130            Ok(a) => a,
131            Err(e) => {
132                println!("cargo:warning=openssl-src: failed to build OpenSSL from source");
133                eprintln!("\n\n\n{e}\n\n\n");
134                std::process::exit(1);
135            }
136        }
137    }

141        let host = &self.host.as_ref().ok_or("HOST dir not set")?[..];
142        let out_dir = self.out_dir.as_ref().ok_or("OUT_DIR not set")?;
143        let build_dir = out_dir.join("build");
144        let install_dir = out_dir.join("install");
145
146        if build_dir.exists() {
147            fs::remove_dir_all(&build_dir).map_err(|e| format!("build_dir: {e}"))?;
148        }
149        if install_dir.exists() {
150            fs::remove_dir_all(&install_dir).map_err(|e| format!("install_dir: {e}"))?;
151        }
152
153        let inner_dir = build_dir.join("src");
154        fs::create_dir_all(&inner_dir).map_err(|e| format!("{}: {e}", inner_dir.display()))?;
155        cp_r(&source_dir(), &inner_dir)?;

157        let perl_program =
158            env::var("OPENSSL_SRC_PERL").unwrap_or(env::var("PERL").unwrap_or("perl".to_string()));
159        let mut configure = Command::new(perl_program);
160        configure.arg("./Configure");
161
162        // Change the install directory to happen inside of the build directory.
163        if host.contains("pc-windows-gnu") {
164            configure.arg(&format!("--prefix={}", sanitize_sh(&install_dir)));
165        } else if host.contains("pc-windows-msvc") || host.contains("win7-windows-msvc") {
166            // On Windows, the prefix argument does not support \ path seperators
167            // when cross compiling.
168            // Always use / as a path seperator instead of \, since that works for both
169            // native and cross builds.
170            configure.arg(&format!(
171                "--prefix={}",
172                install_dir
173                    .to_str()
174                    .ok_or("bad install_dir")?
175                    .replace("\\", "/")
176            ));
177        } else {
178            configure.arg(&format!("--prefix={}", install_dir.display()));
179        }
180
181        // Specify that openssl directory where things are loaded at runtime is
182        // not inside our build directory. Instead this should be located in the
183        // default locations of the OpenSSL build scripts, or as specified by whatever
184        // configured this builder.
185        if target.contains("windows") {
186            configure.arg("--openssldir=SYS$MANAGER:[OPENSSL]");
187        } else {
188            let openssl_dir = self
189                .openssl_dir
190                .as_ref()
191                .ok_or("path to the openssl directory must be set")?;
192            let mut dir_arg: OsString = "--openssldir=".into();
193            dir_arg.push(openssl_dir);
194            configure.arg(dir_arg);
195        }

198            // No shared objects, we just want static libraries
199            .arg("no-shared")
200            .arg("no-module")
201            // Should be off by default on OpenSSL 1.1.0, but let's be extra sure
202            .arg("no-ssl3")
203            // No need to build tests, we won't run them anyway
204            .arg("no-tests")
205            // Nothing related to zlib please
206            .arg("no-comp")
207            .arg("no-zlib")
208            .arg("no-zlib-dynamic")
209            // Avoid multilib-postfix for build targets that specify it
210            .arg("--libdir=lib");

212        if cfg!(feature = "no-dso") {
213            // engine requires DSO support
214            if cfg!(feature = "force-engine") {
215                println!("Feature 'force-engine' requires DSO, ignoring 'no-dso' feature.");
216            } else {
217                configure.arg("no-dso");
218            }
219        }

221        if cfg!(not(feature = "legacy")) {
222            configure.arg("no-legacy");
223        }
224
225        if cfg!(feature = "weak-crypto") {
226            configure
227                .arg("enable-md2")
228                .arg("enable-rc5")
229                .arg("enable-weak-ssl-ciphers");
230        } else {

232                .arg("no-md2")

243        }

255            // right now) but we don't actually need this most of the time.
256            // Disable engine module unless force-engine feature specified
257            if !cfg!(feature = "force-engine") {
258                configure.arg("no-engine");
259            }
260        } else if target.contains("windows") {
261            // We can build the engine feature, but the build doesn't seem
262            // to correctly pick up crypt32.lib functions such as
263            // `__imp_CertOpenStore` when building the capieng engine.
264            // Let's disable just capieng.

270            // stuff depends on, and we don't bind to any of that in any case.

275        // failure (#13), but most other platforms need it for things like
276        // loading system certificates so only disable it on Android.

283            // ASM compiling will be enabled if nasm.exe is installed, unless
284            // the environment variable `OPENSSL_RUST_USE_NASM` is set.

426        // whatnot. Note that this doesn't happen on MSVC b/c things are pretty
427        // different there and this isn't needed most of the time anyway.
428        if !target.contains("msvc") {
429            let mut cc = cc::Build::new();
430            cc.target(target).host(host).warnings(false).opt_level(2);
431            let compiler = cc.get_compiler();
432            let mut cc_env = compiler.cc_env();
433            if cc_env.is_empty() {

437            let path = compiler.path().to_str().ok_or("compiler path")?;

439            // Both `cc::Build` and `./Configure` take into account
440            // `CROSS_COMPILE` environment variable. So to avoid double
441            // prefix, we unset `CROSS_COMPILE` for `./Configure`.
442            configure.env_remove("CROSS_COMPILE");
443
444            let ar = cc.get_archiver();
445            configure.env("AR", ar.get_program());
446            if ar.get_args().count() != 0 {
447                // On some platforms (like emscripten on windows), the ar to use may not be a
448                // single binary, but instead a multi-argument command like `cmd /c emar.bar`.
449                // We can't convey that through `AR` alone, and so also need to set ARFLAGS.

456            // OpenSSL does not support RANLIBFLAGS. Jam the flags in RANLIB.
457            let mut args = vec![ranlib.get_program()];
458            args.extend(ranlib.get_args());

461            // Make sure we pass extra flags like `-ffunction-sections` and
462            // other things like ARM codegen flags.
463            let mut skip_next = false;
464            let mut is_isysroot = false;
465            for arg in compiler.args() {
466                // For whatever reason `-static` on MUSL seems to cause
467                // issues...
468                if target.contains("musl") && arg == "-static" {
469                    continue;
470                }
471
472                // cc includes an `-arch` flag for Apple platforms, but we've
473                // already selected an arch implicitly via the target above, and
474                // OpenSSL contains about the conflict if both are specified.
475                if target.contains("apple") {
476                    if arg == "-arch" {
477                        skip_next = true;
478                        continue;
479                    }
480                }
481
482                // cargo-lipo specifies this but OpenSSL complains
483                if target.contains("apple-ios") {
484                    if arg == "-isysroot" {
485                        is_isysroot = true;
486                        continue;
487                    }
488
489                    if is_isysroot {
490                        is_isysroot = false;
491                        ios_isysroot = Some(arg.to_str().ok_or("isysroot arg")?.to_string());
492                        continue;
493                    }
494                }
495
496                if skip_next {
497                    skip_next = false;
498                    continue;
499                }
500
501                configure.arg(arg);
502            }
503
504            if os.contains("iossimulator") {
505                if let Some(ref isysr) = ios_isysroot {
506                    configure.env(
507                        "CC",
508                        &format!(
509                            "xcrun -sdk iphonesimulator cc -isysroot {}",
510                            sanitize_sh(&Path::new(isysr))
511                        ),
512                    );
513                }
514            }
515
516            if target == "x86_64-pc-windows-gnu" {
517                // For whatever reason OpenSSL 1.1.1 fails to build on
518                // `x86_64-pc-windows-gnu` in our docker container due to an
519                // error about "too many sections". Having no idea what this
520                // error is about some quick googling yields
521                // https://github.com/cginternals/glbinding/issues/135 which
522                // mysteriously mentions `-Wa,-mbig-obj`, passing a new argument
523                // to the assembler. Now I have no idea what `-mbig-obj` does
524                // for Windows nor why it would matter, but it does seem to fix
525                // compilation issues.
526                //
527                // Note that another entirely unrelated issue -
528                // https://github.com/assimp/assimp/issues/177 - was fixed by
529                // splitting a large file, so presumably OpenSSL has a large
530                // file soemwhere in it? Who knows!
531                configure.arg("-Wa,-mbig-obj");
532            }
533
534            if target.contains("pc-windows-gnu") && path.ends_with("-gcc") {
535                // As of OpenSSL 1.1.1 the build system is now trying to execute
536                // `windres` which doesn't exist when we're cross compiling from
537                // Linux, so we may need to instruct it manually to know what
538                // executable to run.
539                let windres = format!("{}-windres", &path[..path.len() - 4]);
540                configure.env("WINDRES", &windres);

543            if target.contains("emscripten") {

545                // `stdatomic.h`, but this doesn't exist on Emscripten. After
546                // reading OpenSSL's source where the error is, we define this
547                // magical (and probably
548                // compiler-internal-should-not-be-user-defined) macro to say
549                // "no atomics are available" and avoid including such a header.
550                configure.arg("-D__STDC_NO_ATOMICS__");
551            }

554                configure.args([
555                    // Termios isn't available whatsoever on WASM/WASI so we disable that
556                    "no-ui-console",
557                    // WASI doesn't support UNIX sockets so we preemptively disable it
558                    "no-sock",
559                    // WASI doesn't have a concept of syslog, so we disable it
560                    "-DNO_SYSLOG",
561                    // WASI doesn't support (p)threads. Disabling preemptively.
562                    "no-threads",
563                    // WASI/WASM aren't really friends with ASM, so we disable it as well.
564                    "no-asm",
565                    // Disables the AFALG engine (AFALG-ENGine)
566                    // Since AFALG depends on `AF_ALG` support on the linux kernel side
567                    // it makes sense that we can't use it.
568                    "no-afalgeng",
569                    "-DOPENSSL_NO_AFALGENG=1",
570                    // wasm lacks signal support; to enable minimal signal emulation, compile with
571                    // -D_WASI_EMULATED_SIGNAL and link with -lwasi-emulated-signal
572                    // The link argument is output in the `Artifacts::print_cargo_metadata` method
573                    "-D_WASI_EMULATED_SIGNAL",
574                    // WASI lacks process-associated clocks; to enable emulation of the `times` function using the wall
575                    // clock, which isn't sensitive to whether the program is running or suspended, compile with
576                    // -D_WASI_EMULATED_PROCESS_CLOCKS and link with -lwasi-emulated-process-clocks
577                    // The link argument is output in the `Artifacts::print_cargo_metadata` method
578                    "-D_WASI_EMULATED_PROCESS_CLOCKS",
579                    // WASI lacks a true mmap; to enable minimal mmap emulation, compile
580                    // with -D_WASI_EMULATED_MMAN and link with -lwasi-emulated-mman
581                    // The link argument is output in the `Artifacts::print_cargo_metadata` method
582                    "-D_WASI_EMULATED_MMAN",
583                    // WASI lacks process identifiers; to enable emulation of the `getpid` function using a
584                    // placeholder value, which doesn't reflect the host PID of the program, compile with
585                    // -D_WASI_EMULATED_GETPID and link with -lwasi-emulated-getpid
586                    // The link argument is output in the `Artifacts::print_cargo_metadata` method
587                    "-D_WASI_EMULATED_GETPID",
588                    // WASI doesn't have chmod right now, so don't try to use it.
589                    "-DNO_CHMOD",
590                ]);
591            }
592
593            if target.contains("musl") {
594                // Hack around openssl/openssl#7207 for now
595                configure.arg("-DOPENSSL_NO_SECURE_MEMORY");
596            }
597        }
598
599        // And finally, run the perl configure script!
600        configure.current_dir(&inner_dir);
601        self.run_command(configure, "configuring OpenSSL build")?;
602
603        // On MSVC we use `nmake.exe` with a slightly different invocation, so
604        // have that take a different path than the standard `make` below.
605        if target.contains("msvc") {
606            let mut build =
607                cc::windows_registry::find(target, "nmake.exe").ok_or("failed to find nmake")?;
608            build.arg("build_libs").current_dir(&inner_dir);
609            self.run_command(build, "building OpenSSL")?;
610
611            let mut install =
612                cc::windows_registry::find(target, "nmake.exe").ok_or("failed to find nmake")?;
613            install.arg("install_dev").current_dir(&inner_dir);
614            self.run_command(install, "installing OpenSSL")?;
615        } else {
616            let mut depend = self.cmd_make()?;
617            depend.arg("depend").current_dir(&inner_dir);
618            self.run_command(depend, "building OpenSSL dependencies")?;
619
620            let mut build = self.cmd_make()?;
621            build.arg("build_libs").current_dir(&inner_dir);
622            if !cfg!(windows) {
623                if let Some(s) = env::var_os("CARGO_MAKEFLAGS") {
624                    build.env("MAKEFLAGS", s);
625                }
626            }
627
628            if let Some(ref isysr) = ios_isysroot {
629                let components: Vec<&str> = isysr.split("/SDKs/").collect();
630                build.env("CROSS_TOP", components[0]);
631                build.env("CROSS_SDK", components[1]);
632            }
633
634            self.run_command(build, "building OpenSSL")?;
635
636            let mut install = self.cmd_make()?;
637            install.arg("install_dev").current_dir(&inner_dir);
638            self.run_command(install, "installing OpenSSL")?;
639        }
640
641        let libs = if target.contains("msvc") {
642            vec!["libssl".to_string(), "libcrypto".to_string()]
643        } else {
644            vec!["ssl".to_string(), "crypto".to_string()]
645        };
646
647        fs::remove_dir_all(&inner_dir).map_err(|e| format!("{}: {e}", inner_dir.display()))?;
648
649        Ok(Artifacts {
650            lib_dir: install_dir.join("lib"),
651            bin_dir: install_dir.join("bin"),
652            include_dir: install_dir.join("include"),
653            libs: libs,
654            target: target.to_string(),
655        })
656    }
657
658    #[track_caller]
659    fn run_command(&self, mut command: Command, desc: &str) -> Result<(), String> {
660        println!("running {:?}", command);
661        let status = command.status();
662
663        let verbose_error = match status {
664            Ok(status) if status.success() => return Ok(()),
665            Ok(status) => format!(
666                "'{exe}' reported failure with {status}",
667                exe = command.get_program().to_string_lossy()
668            ),
669            Err(failed) => match failed.kind() {
670                std::io::ErrorKind::NotFound => format!(
671                    "Command '{exe}' not found. Is {exe} installed?",
672                    exe = command.get_program().to_string_lossy()
673                ),
674                _ => format!(
675                    "Could not run '{exe}', because {failed}",
676                    exe = command.get_program().to_string_lossy()
677                ),
678            },
679        };
680        println!("cargo:warning={desc}: {verbose_error}");
681        Err(format!(
682            "Error {desc}:
683    {verbose_error}
684    Command failed: {command:?}"
685        ))
686    }
687}

689fn cp_r(src: &Path, dst: &Path) -> Result<(), String> {
690    for f in fs::read_dir(src).map_err(|e| format!("{}: {e}", src.display()))? {
691        let f = match f {
692            Ok(f) => f,
693            _ => continue,
694        };
695        let path = f.path();
696        let name = path
697            .file_name()
698            .ok_or_else(|| format!("bad dir {}", src.display()))?;
699
700        // Skip git metadata as it's been known to cause issues (#26) and
701        // otherwise shouldn't be required
702        if name.to_str() == Some(".git") {
703            continue;
704        }
705
706        let dst = dst.join(name);
707        let ty = f.file_type().map_err(|e| e.to_string())?;
708        if ty.is_dir() {
709            fs::create_dir_all(&dst).map_err(|e| e.to_string())?;
710            cp_r(&path, &dst)?;
711        } else if ty.is_symlink() && path.iter().any(|p| p == "cloudflare-quiche") {
712            // not needed to build
713            continue;
714        } else {
715            let _ = fs::remove_file(&dst);
716            if let Err(e) = fs::copy(&path, &dst) {
717                return Err(format!(
718                    "failed to copy '{}' to '{}': {e}",
719                    path.display(),
720                    dst.display()
721                ));
722            }
723        }
724    }
725    Ok(())
726}
727
728fn sanitize_sh(path: &Path) -> String {
729    if !cfg!(windows) {
730        return path.to_string_lossy().into_owned();
731    }
732    let path = path.to_string_lossy().replace("\\", "/");
733    return change_drive(&path).unwrap_or(path);
734
735    fn change_drive(s: &str) -> Option<String> {
736        let mut ch = s.chars();
737        let drive = ch.next().unwrap_or('C');
738        if ch.next() != Some(':') {
739            return None;
740        }
741        if ch.next() != Some('/') {
742            return None;
743        }
744        Some(format!("/{}/{}", drive, &s[drive.len_utf8() + 2..]))
745    }
746}

761    pub fn print_cargo_metadata(&self) {
762        println!("cargo:rustc-link-search=native={}", self.lib_dir.display());
763        for lib in self.libs.iter() {
764            println!("cargo:rustc-link-lib=static={}", lib);
765        }
766        println!("cargo:include={}", self.include_dir.display());
767        println!("cargo:lib={}", self.lib_dir.display());
768        if self.target.contains("windows") {
769            println!("cargo:rustc-link-lib=user32");
770            println!("cargo:rustc-link-lib=crypt32");
771        } else if self.target == "wasm32-wasi" {
772            println!("cargo:rustc-link-lib=wasi-emulated-signal");
773            println!("cargo:rustc-link-lib=wasi-emulated-process-clocks");
774            println!("cargo:rustc-link-lib=wasi-emulated-mman");
775            println!("cargo:rustc-link-lib=wasi-emulated-getpid");
776        }
777    }
